    // ---------- Session ----------
    const me = loadSession();

    // Brand
    document.getElementById("brand").innerHTML = `${logoSVG(44)}<div class="brand-title">FroteBiteMessenger</div>`;

    // Status helpers
    const leftStatus = document.getElementById("leftStatus");

    // ---------- Toast ----------
    const toastHost = document.getElementById("toastHost");
    function showToast(title, text, ttlMs = 3500){
      const el = document.createElement("div");
      el.className = "toast";
      el.innerHTML = `
        <div class="t1">${title}</div>
        <div class="t2">${text}</div>
        <div class="bar"></div>

      toastHost.appendChild(el);
      const bar = el.querySelector(".bar");
      bar.animate([{ transform:"scaleX(1)" }, { transform:"scaleX(0)" }], {
        duration: ttlMs, easing: "linear", fill: "forwards"
      });
      setTimeout(()=> {
        el.classList.add("fade");
        setTimeout(()=> el.remove(), 1500);
      }, ttlMs);
    }

    // ---------- Profile ----------
    const nickPill = document.getElementById("nickPill");

    async function refreshProfile(){
      try{
        const data = await api(`/api/profile?userId=${me.userId}`);
        me.email = data.user.email;
        me.nickname = data.user.nickname || null;
        localStorage.setItem("fbm_user", JSON.stringify(me));
        nickPill.textContent = displayName({ id: me.userId, nickname: me.nickname });
      }catch{
        localStorage.clear();
        location.href = "/login.html";
      }
    }
    await refreshProfile();

    // Modal
    const modalBg = document.getElementById("modalBg");

    const pNotice = document.getElementById("pNotice");

      themeRow.innerHTML = "";

      for(const t of THEMES){

          setTheme(t.id);

    }

    }

    document.getElementById("openProfile").addEventListener("click", openModal);
    document.getElementById("modalClose").addEventListener("click", closeModal);
    modalBg.addEventListener("click", (e)=>{ if(e.target === modalBg) closeModal(); });

    document.getElementById("saveNick").addEventListener("click", async () => {
      const nn = document.getElementById("nickInput").value.trim();

      try{
        const data = await api("/api/profile/nickname", {
          method:"POST",
          headers:{ "Content-Type":"application/json" },
          body: JSON.stringify({ userId: me.userId, nickname: nn })
        });
        me.nickname = data.nickname;
        localStorage.setItem("fbm_user", JSON.stringify(me));
        await refreshProfile();
        openModal();
        pNotice.textContent = "Никнейм обновлён ✅";

    // ---------- Tabs + blur ----------
    const tabChats = document.getElementById("tabChats");
    const tabCalls = document.getElementById("tabCalls");
    const leftView = document.getElementById("leftView");
    const rightView = document.getElementById("rightView");
    const contactsEl = document.getElementById("contacts");
    const callsListEl = document.getElementById("callsList");
    const chatView = document.getElementById("chatView");
    const callsView = document.getElementById("callsView");
    const callControls = document.getElementById("callControls");
    let activeTab = "chats";

    function blurSwitch(toTab){
      leftView.classList.add("blurring");
      rightView.classList.add("blurring");
      setTimeout(()=> {
        activeTab = toTab;
        tabChats.classList.toggle("active", toTab === "chats");
        tabCalls.classList.toggle("active", toTab === "calls");

        contactsEl.style.display = (toTab === "chats") ? "block" : "none";
        callsListEl.style.display = (toTab === "calls") ? "block" : "none";

        chatView.style.display = (toTab === "chats") ? "block" : "none";
        callsView.style.display = (toTab === "calls") ? "block" : "none";

        setTimeout(()=> {
          leftView.classList.remove("blurring");
          rightView.classList.remove("blurring");
        }, 10);
      }, 170);
    }
    tabChats.addEventListener("click", ()=> blurSwitch("chats"));
    tabCalls.addEventListener("click", ()=> blurSwitch("calls"));

    // ---------- Users cache for search + contacts ----------
    let allUsers = [];
    let userById = new Map();
    async function loadUsers(){
      const data = await api("/api/users");
      allUsers = data.users || [];
      userById = new Map(allUsers.map(u => [u.id, u]));
    }
    await loadUsers();

    // ---------- Dialogs (server gives unread + last message) ----------
    let dialogs = []; // from /api/dialogs
    let manualContacts = loadContacts(); // ids added by search
    let contacts = []; // merged ids
    let dialogByOtherId = new Map(); // otherId -> dialog row

    async function refreshDialogs(){
      const data = await api(`/api/dialogs?me=${me.userId}`);
      dialogs = data.dialogs || [];
      dialogByOtherId = new Map(dialogs.map(d => [Number(d.other_id), d]));
    }

    function mergeContacts(){
      const set = new Set();
      for(const d of dialogs) set.add(Number(d.other_id));
      for(const id of manualContacts) set.add(Number(id));
      set.delete(me.userId);

      contacts = Array.from(set).filter(id => userById.has(id)).sort((a,b)=>{
        // sort by dialog last message if exists, else by id
        const da = dialogByOtherId.get(a)?.last_id || 0;
        const db = dialogByOtherId.get(b)?.last_id || 0;
        if(db !== da) return db - da;
        return a - b;
      });
    }

    // ---------- Search/Add contact ----------
    const findEmail = document.getElementById("findEmail");
    const findId = document.getElementById("findId");
    const findMsg = document.getElementById("findMsg");
    let findTimer = null;

    function setFindMessage(t){ findMsg.textContent=t; findMsg.classList.add("show"); }
    function clearFindMessage(){ findMsg.classList.remove("show"); findMsg.textContent=""; }

    function findUser(email, idStr){
      const emailNorm = String(email||"").trim().toLowerCase();
      const idNorm = String(idStr||"").trim();
      if(!emailNorm && !idNorm) return null;

      if(idNorm){
        const id = Number(idNorm);
        if(Number.isFinite(id)){
          const u = userById.get(id);
          if(u) return u;
        }
      }
      if(emailNorm){
        const u = allUsers.find(x => String(x.email).toLowerCase() === emailNorm);
        if(u) return u;
      }
      return null;
    }

    function triggerFind(){
      clearTimeout(findTimer);
      findTimer = setTimeout(async () => {
        if(!findEmail.value.trim() && !findId.value.trim()){
          clearFindMessage();
          return;
        }
        const u = findUser(findEmail.value, findId.value);
        if(!u){
          setFindMessage("Почта или айди введены неверно, никого не найдено");
          return;
        }
        if(u.id === me.userId){
          setFindMessage("Это ваш аккаунт 🙂");
          return;
        }
        clearFindMessage();

        manualContacts = addContact(u.id);
        mergeContacts();
        renderContacts();
        setActiveUser(u.id, true);
      }, 260);
    }
    findEmail.addEventListener("input", triggerFind);
    findId.addEventListener("input", triggerFind);

    // ---------- Contacts render ----------
    let activeUserId = null;
    function stripHtml(html){
      const d = document.createElement("div");
      d.innerHTML = String(html || "");
      return d.textContent || "";
    }

    function renderContacts(){
      contactsEl.innerHTML = "";

      if(contacts.length === 0){
        contactsEl.innerHTML = `<div class="sub" style="padding:10px;">Нет переписок. Добавь человека сверху (почта или id).</div>`;
        document.getElementById("rightTitle").textContent = "Выбери контакт";
        activeUserId = null;
        return;
      }

      for(const id of contacts){
        const u = userById.get(id);
        const d = dialogByOtherId.get(id);

        const unread = Number(d?.unread_count || 0);
        const lastText = d?.last_text ? stripHtml(d.last_text).slice(0, 80) : "Нет сообщений";
        const badge = unread > 0 ? `<div class="unread show">${unread > 99 ? "99+" : unread}</div>` : `<div class="unread"></div>`;

        const item = document.createElement("div");
        item.className = "user-item" + (activeUserId === id ? " active" : "");
        item.innerHTML = `
          <div class="user-row">
            <div class="user-left">
              ${avatarHTML({ id: u.id, nickname: u.nickname }, 34)}
              <div class="user-text">
                <div class="user-title">${displayName({ id: u.id, nickname: u.nickname })}</div>
                <div class="user-sub">${lastText}</div>
              </div>
            </div>
            ${badge}
          </div>
        `;
        item.addEventListener("click", ()=> setActiveUser(id, true));
        contactsEl.appendChild(item);
      }
    }

    // ---------- Thread ----------
    const threadEl = document.getElementById("thread");
    const editor = document.getElementById("editor");

    let replyTarget = null;
    const replyBar = document.getElementById("replyBar");
    const replyTitle = document.getElementById("replyTitle");
    const replyText = document.getElementById("replyText");
    function setReply(target){
      replyTarget = target;
      if(!target){
        replyBar.classList.remove("show");
        return;
      }
      const u = userById.get(target.fromId) || { id: target.fromId, nickname: null };
      replyTitle.textContent = `Ответ ${displayName(u)}`;
      replyText.textContent = target.preview;
      replyBar.classList.add("show");
      editor.focus();
    }
    document.getElementById("replyClear").addEventListener("click", ()=> setReply(null));

    function renderMessages(messages){
      threadEl.innerHTML = "";
      let maxId = 0;

      for(const m of messages){
        maxId = Math.max(maxId, m.id);

        const wrap = document.createElement("div");
        wrap.className = "msg-wrap";
        wrap.dataset.mid = String(m.id);

        const bubble = document.createElement("div");
        bubble.className = "bubble" + (m.sender_id === me.userId ? " me" : "");
        bubble.dataset.mid = String(m.id);

        if(m.reply_to_message_id && m.reply_text){
          const rp = document.createElement("div");
          rp.className = "reply-preview";
          rp.innerHTML = `
            <div class="rt">Ответ ${displayName(userById.get(m.reply_sender_id) || {id:m.reply_sender_id})}</div>
            <div class="rx">${stripHtml(m.reply_text).slice(0, 120)}</div>
          `;
          bubble.appendChild(rp);
        }

        const body = document.createElement("div");
        body.innerHTML = sanitizeRichHtml(m.text);
        bubble.appendChild(body);

        const meta = document.createElement("div");
        meta.className = "meta" + (m.sender_id === me.userId ? " me" : "");
        meta.textContent = new Date(m.created_at).toLocaleString();

        bubble.addEventListener("contextmenu", (e)=>{
          e.preventDefault();
          openCtx(e.clientX, e.clientY, {
            id: m.id,
            senderId: m.sender_id,
            preview: stripHtml(m.text).slice(0, 140)
          });
        });

        wrap.appendChild(bubble);
        wrap.appendChild(meta);
        threadEl.appendChild(wrap);
      }

      threadEl.scrollTop = threadEl.scrollHeight;

      return maxId;
    }

    async function refreshThreadAndRead(){
      if(!activeUserId) return;

      const data = await api(`/api/messages/thread?me=${me.userId}&with=${activeUserId}`);
      const msgs = data.messages || [];
      const maxId = renderMessages(msgs);

      if(maxId > 0){
        await api("/api/read", {
          method:"POST",
          headers:{ "Content-Type":"application/json" },
          body: JSON.stringify({ userId: me.userId, otherId: activeUserId, lastReadMessageId: maxId })
        });
      }
    }

    async function setActiveUser(id, refreshNow){
      activeUserId = Number(id);

      const u = userById.get(activeUserId);
      document.getElementById("rightTitle").textContent = `Диалог: ${displayName({ id: u.id, nickname: u.nickname })}`;

      setReply(null);

      if(refreshNow){
        await refreshThreadAndRead();
        await refreshDialogs();
        mergeContacts();
        renderContacts();
      }

      renderCallControls();
    }

    // Refresh button
    document.getElementById("refresh").addEventListener("click", async ()=> {
      await refreshThreadAndRead();
      await refreshDialogs();
      mergeContacts();
      renderContacts();
    });

    // ---------- Send ----------
    document.getElementById("send").addEventListener("click", async () => {
      if(!activeUserId) return setRightStatus("Выбери контакт");

      const clean = sanitizeRichHtml(editor.innerHTML);
      const plain = stripHtml(clean).trim();
      if(!plain) return setRightStatus("Пустое сообщение");

      try{
        setRightStatus("Отправка...");
        await api("/api/messages/send", {
          method:"POST",
          headers:{ "Content-Type":"application/json" },
          body: JSON.stringify({
            senderId: me.userId,
            receiverId: activeUserId,
            text: clean,
            replyToMessageId: replyTarget?.id || null
          })
        });
        editor.innerHTML = "";
        setReply(null);

        // WS also updates, но на всякий — сразу перечитаем текущий поток
        await refreshThreadAndRead();
        setRightStatus("Ок");
      }catch(e){
        setRightStatus("Ошибка: " + e.message);
      }
    });

    editor.addEventListener("keydown", (e)=>{
      if(e.key === "Enter" && !e.shiftKey){
        e.preventDefault();
        document.getElementById("send").click();
      }
    });

    // ---------- Format toolbar ----------
    const formatbar = document.getElementById("formatbar");
    function selectionInsideEditor(){
      const sel = window.getSelection();
      if(!sel || sel.rangeCount === 0) return false;
      const range = sel.getRangeAt(0);
      return editor.contains(range.commonAncestorContainer);
    }
    function updateFormatbar(){
      const sel = window.getSelection();
      if(!sel || sel.rangeCount === 0 || sel.isCollapsed || !selectionInsideEditor()){
        formatbar.classList.remove("show");
        return;
      }
      const rect = sel.getRangeAt(0).getBoundingClientRect();
      const x = Math.max(10, Math.min(window.innerWidth - 220, rect.left + rect.width/2 - 110));
      const y = Math.max(10, rect.top - 52);
      formatbar.style.left = x + "px";
      formatbar.style.top = y + "px";
      formatbar.classList.add("show");
    }
    document.addEventListener("selectionchange", ()=> {
      if(activeTab !== "chats") return;
      updateFormatbar();
    });
    function cmd(command){
      if(!selectionInsideEditor()) return;
      document.execCommand(command, false, null);
      updateFormatbar();
      editor.focus();
    }
    document.getElementById("fbBold").addEventListener("click", ()=> cmd("bold"));
    document.getElementById("fbItalic").addEventListener("click", ()=> cmd("italic"));
    document.getElementById("fbUnderline").addEventListener("click", ()=> cmd("underline"));
    document.addEventListener("click", (e)=>{
      if(!formatbar.contains(e.target) && e.target !== editor) formatbar.classList.remove("show");
    });

    // ---------- Context menu Reply/Delete ----------
    const ctx = document.getElementById("ctx");
    const ctxReply = document.getElementById("ctxReply");
    const ctxDelete = document.getElementById("ctxDelete");
    let ctxMessage = null;

    function openCtx(x,y,msg){
      ctxMessage = msg;
      ctx.classList.add("show");
      const w = ctx.offsetWidth, h = ctx.offsetHeight;
      ctx.style.left = Math.min(x, window.innerWidth - w - 8) + "px";
      ctx.style.top  = Math.min(y, window.innerHeight - h - 8) + "px";
    }
    function closeCtx(){ ctx.classList.remove("show"); ctxMessage = null; }
    document.addEventListener("click", (e)=>{ if(!ctx.contains(e.target)) closeCtx(); });

    ctxReply.addEventListener("click", ()=>{
      if(!ctxMessage) return;
      setReply({ id: ctxMessage.id, fromId: ctxMessage.senderId, preview: ctxMessage.preview });
      closeCtx();
    });

    function animateDeleteById(mid){
      const el = threadEl.querySelector(`[data-mid="${mid}"]`);
      if(!el) return;
      const h = el.offsetHeight;
      el.style.height = h + "px";
      requestAnimationFrame(()=>{
        el.classList.add("deleting");
        setTimeout(()=>{
          el.classList.add("collapse");
          el.style.height = "0px";
          setTimeout(()=> el.remove(), 260);
        }, 190);
      });
    }

    ctxDelete.addEventListener("click", async ()=>{
      if(!ctxMessage) return;
      const mid = ctxMessage.id;
      closeCtx();
      try{
        setRightStatus("Удаление...");
        await api(`/api/messages/${mid}?requesterId=${me.userId}`, { method:"DELETE" });
        animateDeleteById(mid);
        setRightStatus("Удалено");
      }catch(e){
        setRightStatus("Ошибка: " + e.message);
      }
    });

    // ---------- Calls (unchanged local) ----------
    const CALLS_KEY = "fbm_calls";
    function loadCalls(){ try { return JSON.parse(localStorage.getItem(CALLS_KEY)||"[]"); } catch { return []; } }
    function saveCalls(list){ localStorage.setItem(CALLS_KEY, JSON.stringify(list)); }

    function renderCallsList(){
      const calls = loadCalls();
      callsListEl.innerHTML = "";
      if(calls.length === 0){
        callsListEl.innerHTML = `<div class="sub">Пока нет звонков.</div>`;
        return;
      }
      const sorted = [...calls].sort((a,b)=> b.ts - a.ts);
      for(const c of sorted){
        const u = userById.get(c.withId) || { id: c.withId, nickname: null };
        const el = document.createElement("div");
        el.className = "user-item";
        el.style.cursor = "default";
        el.innerHTML = `
          <div class="user-row">
            <div class="user-left">
              ${avatarHTML({ id: u.id, nickname: u.nickname }, 34)}
              <div class="user-text">
                <div class="user-title">${c.type === "out" ? "Исходящий" : "Входящий"} • ${displayName(u)}</div>
                <div class="user-sub">${new Date(c.ts).toLocaleString()}</div>
              </div>
            </div>
          </div>
        `;
        callsListEl.appendChild(el);
      }
    }

    function renderCallControls(){
      callControls.innerHTML = "";
      if(!activeUserId){
        callControls.innerHTML = `<div class="sub">Выбери контакт слева, чтобы появилась кнопка “Позвонить”.</div>`;
        return;
      }
      const u = userById.get(activeUserId);
      const btn = document.createElement("button");
      btn.className = "btn";
      btn.textContent = "Позвонить";
      btn.addEventListener("click", ()=>{
        const calls = loadCalls();
        calls.push({ type:"out", withId: activeUserId, ts: Date.now() });
        saveCalls(calls);
        showToast("Звонок", `Исходящий вызов: ${displayName(u)}`, 2500);
        renderCallsList();
        blurSwitch("calls");
      });
      callControls.appendChild(btn);
    }

    // ---------- WebSocket real-time ----------
    let ws = null;
    let wsReady = false;

    function wsUrl(){
      const proto = location.protocol === "https:" ? "wss" : "ws";
      return `${proto}://${location.host}/ws?userId=${me.userId}`;
    }

    function connectWS(){
      wsReady = false;
      ws = new WebSocket(wsUrl());

      ws.onopen = () => { wsReady = true; };
      ws.onclose = () => { wsReady = false; setTimeout(connectWS, 1200); };

      ws.onmessage = async (evt) => {
        let msg = null;
        try { msg = JSON.parse(evt.data); } catch { return; }

        if(msg.type === "profile:update"){
          // refresh users cache for nice names
          await loadUsers();
          await refreshDialogs();
          mergeContacts();
          renderContacts();
          if(activeUserId){
            const u = userById.get(activeUserId);
            if(u) document.getElementById("rightTitle").textContent = `Диалог: ${displayName(u)}`;
          }
          await refreshProfile();
          return;
        }

        if(msg.type === "message:new"){
          const m = msg.message;
          const otherId = (m.sender_id === me.userId) ? m.receiver_id : m.sender_id;

          // update dialogs + UI
          await refreshDialogs();
          mergeContacts();
          renderContacts();

          // if current dialog open -> refresh thread + mark read
          if(activeUserId && otherId === activeUserId){
            await refreshThreadAndRead();
          } else {
            // toast only for incoming
            if(m.sender_id !== me.userId){
              const u = userById.get(otherId) || { id: otherId, nickname: null };
              showToast("Новое сообщение", `${displayName(u)}: ${stripHtml(m.text).slice(0, 120)}`);
            }
          }
          return;
        }

        if(msg.type === "message:delete"){
          const mid = Number(msg.messageId);
          // remove from UI if visible
          const el = threadEl.querySelector(`[data-mid="${mid}"]`);
          if(el) animateDeleteById(mid);
          await refreshDialogs();
          mergeContacts();
          renderContacts();
          return;
        }

        if(msg.type === "read:update"){
          // another tab of mine updated reads -> refresh contacts badge
          await refreshDialogs();
          mergeContacts();
          renderContacts();
          return;
        }
      };
    }

    connectWS();

    // ---------- Init ----------
    await refreshDialogs();
    mergeContacts();
    renderContacts();
    renderCallsList();
    renderCallControls();

    // auto open first
    if(contacts.length > 0){
      await setActiveUser(contacts[0], true);
    }

    // small fallback polling (in case ws blocked)
    setInterval(async ()=>{
      if(wsReady) return;
      await refreshDialogs();
      mergeContacts();
      renderContacts();
      if(activeUserId && activeTab === "chats"){
        await refreshThreadAndRead();
      }
    }, 2500);
  </script>
